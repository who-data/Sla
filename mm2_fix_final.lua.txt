local function ensureExecutorFS()

        if type(fn) ~= "function" then

                error("HoneyLua requires executor file APIs. Missing: " .. name) 
            end 
        end

local function decodeJson(p, f) 
    local ok, d = pcall(function() return HttpService:JSONDecode(p) end)
    return (ok and type(d) == "table") and d or f 
end

local function encodeJson(p) 
    return HttpService:JSONEncode(p) 
end

local function readSettings() 
    return FS.isfile(SETTINGS_FILE) and decodeJson(FS.readfile(SETTINGS_FILE), DEFAULT_SETTINGS) or DEFAULT_SETTINGS 
end

local function writeSettings(s) 
    FS.writefile(SETTINGS_FILE, encodeJson(s)) 
end

local function configPath(n) 
    return CONFIG_FOLDER .. "/" .. n .. ".json" 
end

local function readConfig(n) 
    local p = configPath(n)
    return FS.isfile(p) and decodeJson(FS.readfile(p), {}) or {} 
end

local function writeConfig(n, d) 
    FS.writefile(configPath(n), encodeJson(d)) 
end

local function resetConfig(n) 
    writeConfig(n, {})
    return {} 
end

local function deleteConfigFile(n) 
    if FS.delfile and FS.isfile(configPath(n)) then 
        pcall(function() FS.delfile(configPath(n)) end) 
    end 
end

local settings = readSettings()
for k, v in pairs(DEFAULT_SETTINGS) do 
    if settings[k] == nil then 
        settings[k] = v 
    end 
end
if not settings.defaultConfigs then settings.defaultConfigs = {} end

local currentConfigName = settings.defaultConfigs[gameId] or "default"
local configData = readConfig(currentConfigName)

local function syncSettings() 
    writeSettings(settings) 
end

local function getAccentColors()
    return ACCENT_COLORS[settings.accentColor] or ACCENT_COLORS.blue
end

local function listConfigs()
    local configs = {}
    if FS.isfolder(CONFIG_FOLDER) then
        for _, fp in ipairs(FS.listfiles(CONFIG_FOLDER)) do
            local name = fp:match("([^/\\]+)%.json$")
            if name then table.insert(configs, name) end
        end
    end
    if #configs == 0 then table.insert(configs, "default") end
    table.sort(configs)
    return configs
end

local function applyConfigToControls()
    for key, ctrl in pairs(Controls) do
        local val = configData[key]
        if val == nil then val = ctrl.default end
        ctrl.set(val, false)
    end
end

local function captureControlValues()
    local data = {}
    for key, ctrl in pairs(Controls) do 
        data[key] = ctrl.get() 
    end
    return data
end

local function createConfig(n) 
    currentConfigName = n
    configData = {}
    writeConfig(currentConfigName, configData)
    syncSettings()
    applyConfigToControls()
    updateStatus("Created: " .. currentConfigName) 
end

local function saveConfig() 
    configData = captureControlValues()
    writeConfig(currentConfigName, configData)
    syncSettings()
    updateStatus("Saved: " .. currentConfigName) 
end

local function loadConfig(n) 
    currentConfigName = n
    configData = readConfig(currentConfigName)
    syncSettings()
    applyConfigToControls()
    updateStatus("Loaded: " .. currentConfigName) 
end

local function resetCurrentConfig() 
    configData = resetConfig(currentConfigName)
    applyConfigToControls()
    updateStatus("Reset: " .. currentConfigName) 
end

local function registerControl(id, def, getter, setter)
    Controls[id] = {default = def, get = getter, set = setter}
    local val = configData[id]
    if val == nil then val = def end
    setter(val, true)
end

local function setTabActive(tabName)
    Sections = {}
    for name, tab in pairs(Tabs) do
        local isActive = name == tabName
        local colors = getAccentColors()
        local targetBg = isActive and Color3.fromRGB(30, 30, 35) or Color3.fromRGB(20, 20, 25)
        local targetText = isActive and colors.primary or Color3.fromRGB(140, 140, 150)
        
        TweenService:Create(tab.button, TweenInfo.new(0.2, Enum.EasingStyle.Quint), {
            BackgroundColor3 = targetBg, 
            TextColor3 = targetText
        }):Play()
        
        if tab.indicator then
            local targetTrans = isActive and 0 or 1
            TweenService:Create(tab.indicator, TweenInfo.new(0.2, Enum.EasingStyle.Quint), {
                BackgroundTransparency = targetTrans
            }):Play()
        end

        if isActive and not tab.container.Visible then
            tab.container.Visible = true
            tab.container.GroupTransparency = 1
            TweenService:Create(tab.container, TweenInfo.new(0.3, Enum.EasingStyle.Quint), {
                GroupTransparency = 0
            }):Play()
            task.defer(function()
                if tab.api.FirstSection then 
                    updateSection(tab.api.FirstSection) 
                end
            end)
        elseif not isActive and tab.container.Visible then
            local tw = TweenService:Create(tab.container, TweenInfo.new(0.2, Enum.EasingStyle.Quint), {
                GroupTransparency = 1
            })
            tw:Play()
            tw.Completed:Connect(function() 
                if ActiveTab ~= name then 
                    tab.container.Visible = false 
                end 
            end)
        end
    end
    ActiveTab = tabName
end

local function createModernBackground(parent)
    local bg = Instance.new("Frame")
    bg.Name = "ModernBG"
    bg.Size = UDim2.new(1, 0, 1, 0)
    bg.BackgroundTransparency = 1
    bg.ClipsDescendants = true
    bg.ZIndex = 1
    bg.Parent = parent
    UI.particleContainer = bg

    AnimatedBgEnabled = settings.animatedBg
    if not settings.animatedBg then 
        bg.Visible = false 
    end

    -- Floating particles effect
    for i = 1, 25 do
        local particle = Instance.new("Frame")
        particle.BackgroundColor3 = getAccentColors().primary
        particle.BackgroundTransparency = 0.8
        particle.BorderSizePixel = 0
        particle.Size = UDim2.new(0, math.random(2, 5), 0, math.random(2, 5))
        particle.Position = UDim2.new(math.random(), 0, math.random(), 0)
        particle.ZIndex = 1
        particle.Parent = bg
        
        local corner = Instance.new("UICorner")
        corner.CornerRadius = UDim.new(1, 0)
        corner.Parent = particle

        task.spawn(function()
            task.wait(i * 0.3)
            while particle and particle.Parent do
                if not AnimatedBgEnabled then
                    task.wait(0.5)
                    continue
                end
                
                local newX = math.random()
                local newY = math.random()
                local duration = 8 + math.random() * 8
                
                local colors = getAccentColors()
                particle.BackgroundColor3 = colors.primary
                
                TweenService:Create(particle, TweenInfo.new(duration, Enum.EasingStyle.Sine), {
                    Position = UDim2.new(newX, 0, newY, 0),
                    BackgroundTransparency = 0.7 + math.random() * 0.2
                }):Play()
                
                task.wait(duration)
            end
        end)
    end
    
    return bg
end

local function enableDragging(frame, handle)
    local dragging, dragStart, startPos = false, nil, nil
    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then 
                    dragging = false 
                end
            end)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement and dragging then
            local delta = input.Position - dragStart
            frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
end

local function enableResizing(frame, handle)
    local resizing, startSize, startMouse = false, nil, nil
    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            resizing = true
            startMouse = Vector2.new(input.Position.X, input.Position.Y)
            startSize = frame.Size
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 and resizing then
            resizing = false
            settings.uiWidth = frame.Size.X.Offset
            settings.uiHeight = frame.Size.Y.Offset
            writeSettings(settings)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if resizing and input.UserInputType == Enum.UserInputType.MouseMovement then

local isToggling = false
function UI.ToggleUI(state)
    if isToggling then return end
    isToggling = true
    
    local target = state ~= nil and state or not UI.mainFrame.Visible
    local uiW, uiH = settings.uiWidth or 520, settings.uiHeight or 380
    
    if target then
        UI.mainFrame.Visible = true
        UI.mainFrame.Size = UDim2.fromOffset(uiW * 0.95, uiH * 0.95)
        UI.mainStroke.Transparency = 1
        
        if UI.mainFrame:IsA("CanvasGroup") then
            UI.mainFrame.GroupTransparency = 1
            TweenService:Create(UI.mainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Quint), {
                GroupTransparency = 0
            }):Play()
        end
        
        TweenService:Create(UI.mainFrame, TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
            Size = UDim2.fromOffset(uiW, uiH)
        }):Play()
        
        local tw = TweenService:Create(UI.mainStroke, TweenInfo.new(0.3, Enum.EasingStyle.Quint), {
            Transparency = 0.5
        })
        tw:Play()
        tw.Completed:Wait()
    else
        TweenService:Create(UI.mainFrame, TweenInfo.new(0.25, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
            Size = UDim2.fromOffset(uiW * 0.95, uiH * 0.95)
        }):Play()
        
        if UI.mainFrame:IsA("CanvasGroup") then
            TweenService:Create(UI.mainFrame, TweenInfo.new(0.2, Enum.EasingStyle.Quint), {
                GroupTransparency = 1
            }):Play()
        end
        
        local tw = TweenService:Create(UI.mainStroke, TweenInfo.new(0.2, Enum.EasingStyle.Quint), {
            Transparency = 1
        })
        tw:Play()
        tw.Completed:Wait()
        UI.mainFrame.Visible = false
    end
    
    isToggling = false
end

                    }):Play()
                end)
                ib.MouseLeave:Connect(function()

                        BackgroundColor3 = Color3.fromRGB(45, 45, 50),
                        TextColor3 = Color3.fromRGB(220, 220, 230)
                    }):Play()
                end)
                ib.MouseButton1Click:Connect(function()
                    sel(it)
                    close()
                end)
                table.insert(itemBtns, ib)
            end
        end
        
        createButtons()
        sel(selected, true)
        
        registerControl(options.id, selected, function() return selected end, function(v, silent) sel(v, silent) end)
        
        local dropdownApi = {label = lbl, dropdown = dd, frame = row}
        
        function dropdownApi:SetItems(newItems)
            items = newItems or {}
            createButtons()
            if not table.find(items, selected) then
                sel(items[1] or "None")
            end
            if isOpen then
                TweenService:Create(lf, TweenInfo.new(0.15), {
                    Size = UDim2.new(1, 0, 0, math.min(#items * 34 + 12, 200))
                }):Play()
            end
        end
        
        function dropdownApi:SetValue(value, silent)
            if table.find(items, value) then
                sel(value, silent)
            end
        end
        
        function dropdownApi:GetValue()
            return selected
        end
        
        return dropdownApi
    end

    local function addToggleSlider(options)
        local row = Instance.new("Frame")

        knob.ZIndex = 4
        knob.Parent = tf
        Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

        local function applyT(v, a)
            state = v
            local tP = state and UDim2.fromOffset(24, 2) or UDim2.fromOffset(2, 2)
            local tC = state and getAccentColors().primary or Color3.fromRGB(55, 55, 60)
            local kC = state and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(180, 180, 190)
            if a then
                TweenService:Create(knob, TweenInfo.new(0.2), {Position = tP, BackgroundColor3 = kC}):Play()
                TweenService:Create(tf, TweenInfo.new(0.2), {BackgroundColor3 = tC}):Play()
            else
                knob.Position = tP
                knob.BackgroundColor3 = kC
                tf.BackgroundColor3 = tC
            end
        end
        applyT(state, false)
        
        local tBtn = Instance.new("TextButton")
        tBtn.Size = UDim2.new(0, 60, 1, 0)
        tBtn.Position = UDim2.new(1, -60, 0, 0)
        tBtn.BackgroundTransparency = 1
        tBtn.Text = ""
        tBtn.ZIndex = 5
        tBtn.Parent = row
        tBtn.MouseButton1Click:Connect(function()
            applyT(not state, true)
            if options.onChangeT then options.onChangeT(state) end
        end)
        
        -- Slider
        local slCont = Instance.new("Frame")
        slCont.Size = UDim2.new(1, -20, 0, 28)
        slCont.Position = UDim2.new(0, 16, 0.6, 0)
        slCont.BackgroundTransparency = 1
        slCont.ZIndex = 3
        slCont.Parent = row
        
        local sliderBg = Instance.new("Frame")
        sliderBg.Size = UDim2.new(1, -70, 0, 4)
        sliderBg.Position = UDim2.new(0, 0, 0.5, -2)
        sliderBg.BackgroundColor3 = Color3.fromRGB(55, 55, 60)
        sliderBg.BorderSizePixel = 0
        sliderBg.ZIndex = 4
        sliderBg.Parent = slCont
        Instance.new("UICorner", sliderBg).CornerRadius = UDim.new(1, 0)
        
        local fill = Instance.new("Frame")
        fill.Size = UDim2.new(0, 0, 1, 0)
        fill.BackgroundColor3 = getAccentColors().primary
        fill.BorderSizePixel = 0
        fill.ZIndex = 5
        fill.Parent = sliderBg
        Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)
        
        local thumb = Instance.new("Frame")
        thumb.Size = UDim2.fromOffset(12, 12)
        thumb.Position = UDim2.new(0, -6, 0.5, -6)
        thumb.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        thumb.ZIndex = 6
        thumb.Parent = sliderBg
        Instance.new("UICorner", thumb).CornerRadius = UDim.new(1, 0)
        
        local valInput = Instance.new("TextBox")
        valInput.Size = UDim2.fromOffset(45, 24)
        valInput.Position = UDim2.new(1, -50, 0.5, -12)
        valInput.BackgroundColor3 = Color3.fromRGB(40, 40, 45)
        valInput.BorderSizePixel = 0
        valInput.Font = Enum.Font.Gotham
        valInput.TextSize = 10
        valInput.TextColor3 = getAccentColors().primary
        valInput.Text = ""
        valInput.ZIndex = 5
        valInput.Parent = slCont
        Instance.new("UICorner", valInput).CornerRadius = UDim.new(0, 6)
        
        local min, max, step, value = options.min or 0, options.max or 100, options.step or 1, options.defaultS or options.min or 0
        
        local function setVal(nv, ut, silent)
            local c = math.clamp(nv, min, max)
            local s = math.floor((c - min) / step + 0.5) * step + min
            value = s
            local a = (value - min) / (max - min)
            if ut then
                TweenService:Create(fill, TweenInfo.new(0.1), {Size = UDim2.new(a, 0, 1, 0)}):Play()
                TweenService:Create(thumb, TweenInfo.new(0.1), {Position = UDim2.new(a, -6, 0.5, -6)}):Play()
            else
                fill.Size = UDim2.new(a, 0, 1, 0)
                thumb.Position = UDim2.new(a, -6, 0.5, -6)
            end
            valInput.Text = tostring(value)
            if not silent and options.onChangeS then options.onChangeS(value) end
        end
        
        valInput.FocusLost:Connect(function()
            local nv = tonumber(valInput.Text)
            if nv then setVal(nv, true, false) else valInput.Text = tostring(value) end
        end)
        
        local dragging = false
        local sBtn = Instance.new("TextButton")
        sBtn.Size = UDim2.new(1, 0, 2, 10)
        sBtn.Position = UDim2.new(0, 0, -0.5, -5)
        sBtn.BackgroundTransparency = 1
        sBtn.Text = ""
        sBtn.ZIndex = 10
        sBtn.Parent = sliderBg
        
        sBtn.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = true end end)
        UserInputService.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end end)
        UserInputService.InputChanged:Connect(function(i) 
            if dragging and i.UserInputType == Enum.UserInputType.MouseMovement then 
                local r = math.clamp((i.Position.X - sliderBg.AbsolutePosition.X) / sliderBg.AbsoluteSize.X, 0, 1)
                setVal(min + (max - min) * r, false, false)
            end 
        end)
        
        setVal(value, false, true)
        registerControl(options.idT, options.defaultT or false, function() return state end, function(v, silent) applyT(v, false); if not silent and options.onChangeT then options.onChangeT(v) end end)
        registerControl(options.idS, options.defaultS or min, function() return value end, function(v, silent) setVal(v, false, silent) end)
        return {row = row}
    end

    api = {
        AddToggle = addToggle,
        AddButton = addButton,
        AddSlider = addSlider,
        AddToggleSlider = addToggleSlider,
        AddDropdown = addDropdown,

        AddText = addText,

    tabBtn.MouseButton1Click:Connect(function() setTabActive(tabName) end)
    if not ActiveTab then setTabActive(tabName) end
    return api
end
