local function DeepCopy(orig)
    local copy = {}
    if type(orig) == "table" then
        for k, v in next, orig, nil do copy[DeepCopy(k)] = DeepCopy(v) end
    else copy = orig end
    return copy
end

local Config = DeepCopy(DefaultConfig)

-- SAVE / LOAD SIMPLES E CONFIÁVEL
-- ============================================================
local function SaveConfig()
    if not writefile then return end
    pcall(function()
        writefile(ConfigFileName, HttpService:JSONEncode(Config))
    end)
end

local function LoadConfig()
    if not (isfile and isfile(ConfigFileName)) then return end
    local ok, decoded = pcall(function()
        return HttpService:JSONDecode(readfile(ConfigFileName))
    end)
    if ok and decoded and type(decoded) == "table" then
        for k, v in pairs(decoded) do
            if Config[k] ~= nil then
                Config[k] = v
            end
        end
    end
end
LoadConfig()

local function ScheduleSave()
    if _savePending then return end
    _savePending = true
    task.delay(1, function()
        _savePending = false
        SaveConfig()
    end)
end

local function MakeDraggableCornersOnly(frame)
    local dragging, dragInput, dragStart, startPos
    local cs = 20
    local function mkCorner(pos)
        local d = Instance.new("Frame")
        d.Size = UDim2.new(0, cs, 0, cs)
        d.BackgroundTransparency = 1
        d.ZIndex = frame.ZIndex + 1
        d.Parent = frame
        if pos == "TL" then d.Position = UDim2.new(0,0,0,0)
        elseif pos == "TR" then d.Position = UDim2.new(1,-cs,0,0)
        elseif pos == "BL" then d.Position = UDim2.new(0,0,1,-cs)
        elseif pos == "BR" then d.Position = UDim2.new(1,-cs,1,-cs) end
        return d
    end
    local detectors = { mkCorner("TL"), mkCorner("TR"), mkCorner("BL"), mkCorner("BR") }
    local function startDrag(input)
        dragging = true; dragStart = input.Position; startPos = frame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then dragging = false end
        end)
    end
    for _, d in ipairs(detectors) do
        d.InputBegan:Connect(function(i)
            if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
                startDrag(i)
            end
        end)
        d.InputChanged:Connect(function(i)
            if i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch then
                dragInput = i
            end
        end)
    end
    UserInputService.InputChanged:Connect(function(i)
        if dragging and i == dragInput then
            local delta = i.Position - dragStart
            frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset+delta.X, startPos.Y.Scale, startPos.Y.Offset+delta.Y)
        end
    end)
end

    end
end

-- GetLeadShotPosition: prediction manual + ping comp + offsets XYZ de câmera
local function GetLeadShotPosition(targetRootPart)
    local pos = targetRootPart.Position
    local vel = targetRootPart.Velocity

    -- Tempo total de avanço = prediction manual + ping (se PingComp ativo)
    local totalTime = Config.Prediction
    if Config.PingComp then
        local ok, pingMs = pcall(function()
            return Stats.Network.ServerStatsItem["Data Ping"]:GetValue()
        end)
        totalTime = totalTime + (ok and pingMs or 80) / 1000
    end
    if totalTime > 0 then
        pos = pos + vel * totalTime
    end

    -- Offsets em espaço local do personagem alvo (Right/Up/Look do HRP)
    if Config.OffsetX ~= 0 or Config.OffsetY ~= 0 or Config.OffsetZ ~= 0 then
        local hrpCF = targetRootPart.CFrame
        pos = pos
            + hrpCF.RightVector   * Config.OffsetX
            + hrpCF.UpVector      * Config.OffsetY
            + hrpCF.LookVector    * Config.OffsetZ
    end

    return pos
end

-- Wallcheck reutilizável
local function HasLineOfSight(origin, targetChar, targetPos)
    local rp = RaycastParams.new()
    rp.FilterDescendantsInstances = {LocalPlayer.Character, targetChar,
        workspace:FindFirstChild("GunDrop")}
    rp.FilterType = Enum.RaycastFilterType.Exclude
    return not workspace:Raycast(origin, (targetPos - origin), rp)
end

local function FindGun()
    local char = LocalPlayer.Character
    if char then
        for _, i in ipairs(char:GetChildren()) do
            if i:IsA("Tool") and (string.find(string.lower(i.Name),"gun") or i.Name=="Revolver") then return i end
        end
    end
    for _, i in ipairs(LocalPlayer.Backpack:GetChildren()) do
        if i:IsA("Tool") and (string.find(string.lower(i.Name),"gun") or i.Name=="Revolver") then return i end
    end
    return nil
end

-- FireGunAt: usado pelo AutoShoot/AutoKill (sem equip bugado)
local function FireGunAt(targetChar)
    local gun = FindGun()
    if not gun then return false end
    local char = LocalPlayer.Character
    if not char then return false end
    local hrp = targetChar:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    local wasInBackpack = (gun.Parent == LocalPlayer.Backpack)
    if wasInBackpack then char.Humanoid:EquipTool(gun); task.wait(0.05) end
    local tPos = GetLeadShotPosition(hrp)
    if gun:FindFirstChild("Shoot") then
        pcall(function()
            gun.Shoot:FireServer(
                CFrame.new(Camera.CFrame.Position, tPos),
                CFrame.new(tPos)
            )
        end)
    end
    if wasInBackpack and gun.Parent == char then gun.Parent = LocalPlayer.Backpack end
    return true
end

-- Botão NEXUS AIM: atira no murderer diretamente via FireServer
local function PerformSilentAimDirectly()
    if not Config.SilentAim then return false end

    local gun = nil
    local char = LocalPlayer.Character
    if char then
        for _, i in ipairs(char:GetChildren()) do
            if i:IsA("Tool") and (string.find(string.lower(i.Name),"gun") or i.Name=="Revolver") then
                gun = i; break
            end
        end
    end
    if not gun then
        for _, i in ipairs(LocalPlayer.Backpack:GetChildren()) do
            if i:IsA("Tool") and (string.find(string.lower(i.Name),"gun") or i.Name=="Revolver") then
                gun = i; break
            end
        end
    end
    if not gun then return false end

    local targetPlr = nil
    for plrName, role in pairs(RoleCache) do
        if role == "Murderer" then
            local p = Players:FindFirstChild(plrName)
            if p and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                targetPlr = p; break
            end
        end
    end
    if not targetPlr then return false end

    if Config.WallCheck then
        local origin = Camera.CFrame.Position
        local tgt    = targetPlr.Character.HumanoidRootPart.Position
        local rp = RaycastParams.new()
        rp.FilterDescendantsInstances = {LocalPlayer.Character, targetPlr.Character,
            workspace:FindFirstChild("GunDrop")}
        rp.FilterType = Enum.RaycastFilterType.Exclude
        if workspace:Raycast(origin, (tgt - origin), rp) then return false end
    end

    local wasInBackpack = (gun.Parent == LocalPlayer.Backpack)
    if wasInBackpack then
        LocalPlayer.Character.Humanoid:EquipTool(gun)
        task.wait(0.1)
    end

    local tRoot = targetPlr.Character.HumanoidRootPart
    local tPos  = GetLeadShotPosition(tRoot)

    -- BulletTP é tratado pelo hook do FireServer em hookBulletTP
    -- PerformSilentAimDirectly só dispara normalmente — o hook intercepta e modifica os args
    if gun:FindFirstChild("Shoot") then
        gun.Shoot:FireServer(
            CFrame.new(Camera.CFrame.Position, tPos),
            CFrame.new(tPos)
        )
        if wasInBackpack and gun.Parent == LocalPlayer.Character then
            gun.Parent = LocalPlayer.Backpack
        end
        return true
    end
    return false
end

local function getGunTarget()
    if not Config.GunSilentAim then return nil end

    for plrName, role in pairs(RoleCache) do
        if role == "Murderer" then
            local plr = Players:FindFirstChild(plrName)
            if plr and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                local hum = plr.Character:FindFirstChild("Humanoid")
                if hum and hum.Health > 0 then
                    return plr.Character.HumanoidRootPart
                end
            end
        end
    end
    return nil
end

-- KnifeSilentAim: retorna HRP mais próximo ao centro da tela, excluindo murderer
local function hasKnife(player)
    if not player or not player.Character then return false end
    for _, item in pairs(player.Backpack:GetChildren()) do
        if item:IsA("Tool") then
            local n = item.Name:lower()
            if n:find("knife") or n:find("fac") or n=="knife" or n=="murder knife" then return true end
        end
    end
    for _, item in pairs(player.Character:GetChildren()) do
        if item:IsA("Tool") then
            local n = item.Name:lower()
            if n:find("knife") or n:find("fac") or n=="knife" or n=="murder knife" then return true end
        end
    end
    return false
end

local function detectMurder()
    while true do
        task.wait(1)
        local found = nil
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and hasKnife(p) then found = p; break end
        end
        murder = found
    end
end
task.spawn(detectMurder)

local function getclosestplayer()
    if not Config.KnifeSilentAim then return nil end
    if not knifeEquipped then return nil end

    local vp     = Camera.ViewportSize
    local center = Vector2.new(vp.X / 2, vp.Y / 2)
    local best, bestDist = nil, math.huge

    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p ~= murder and p.Character then
            local hrp = p.Character:FindFirstChild("HumanoidRootPart")
            local hum = p.Character:FindFirstChild("Humanoid")
            if hrp and hum and hum.Health > 0 then
                local sp, onScreen = Camera:WorldToViewportPoint(hrp.Position)
                if onScreen then
                    local dist = (Vector2.new(sp.X, sp.Y) - center).Magnitude
                    if dist < bestDist then
                        bestDist = dist; best = hrp
                    end
                end
            end
        end
    end
    return best
end

-- hookKnifeThrown: redireciona o evento KnifeThrown para o alvo do knife SA
local originalKnifeThrown = nil
local function hookKnifeThrown()
    local character = LocalPlayer.Character
    if not character then return end
    local knifeTool = character:FindFirstChild("Knife") or LocalPlayer.Backpack:FindFirstChild("Knife")
    if not knifeTool then return end
    local knifeEvents = knifeTool:FindFirstChild("Events")
    if not knifeEvents then return end
    local knifeThrownEvent = knifeEvents:FindFirstChild("KnifeThrown")
    if not knifeThrownEvent then return end
    if not originalKnifeThrown then
        originalKnifeThrown = knifeThrownEvent.FireServer
    end
    knifeThrownEvent.FireServer = function(self, ...)
        local args = {...}
        if closestKnifePart and knifeEquipped and Config.KnifeSilentAim then
            local targetPos    = closestKnifePart.Position
            local characterPos = character:GetPivot().Position
            local direction    = (targetPos - characterPos).Unit
            local newCFrame    = CFrame.new(characterPos, characterPos + direction)
            if #args >= 1 then args[1] = newCFrame end
        end
        return originalKnifeThrown(self, unpack(args))
    end
end

-- ============================================================
-- BULLET TP
-- Quando gun dispara, TP do HRP até o murderer por 1 frame
-- O Raycast do servidor parte do murderer → hit garantido
-- ============================================================
local _bulletTPConns  = {}
local _bulletTPHooks  = {} -- guarda os hooks de FireServer por gun

local function GetMurderPos()
    for plrName, role in pairs(RoleCache) do
        if role == "Murderer" then
            local p = Players:FindFirstChild(plrName)
            if p and p.Character then
                local hrp = p.Character:FindFirstChild("HumanoidRootPart")
                if hrp then return hrp.Position, hrp.CFrame end
            end
        end
    end
    return nil, nil
end

local function hookBulletTP(gun)
    if not gun then return end

    -- desconecta hook antigo
    if _bulletTPConns[gun] then
        _bulletTPConns[gun]:Disconnect()
        _bulletTPConns[gun] = nil
    end

    -- restaura FireServer original se já tiver hookado
    if _bulletTPHooks[gun] then
        local shootEvt = gun:FindFirstChild("Shoot")
        if shootEvt and _bulletTPHooks[gun].orig then
            pcall(function()
                hookfunction(shootEvt.FireServer, _bulletTPHooks[gun].orig)
            end)
        end
        _bulletTPHooks[gun] = nil
    end

    local shootEvt = gun:FindFirstChild("Shoot")
    if not shootEvt then return end

    -- Hook no FireServer do RemoteEvent pra interceptar args reais do jogo
    local origFire = shootEvt.FireServer
    _bulletTPHooks[gun] = { orig = origFire }

    local ok = pcall(function()
        hookfunction(origFire, newcclosure(function(self, ...)
            if not Config.BulletTP then
                return origFire(self, ...)
            end

            local mode    = Config.BulletTPMode or "End"
            local mPos, mCF = GetMurderPos()
            if not mPos then
                return origFire(self, ...)
            end

            local args = {...}

            -- Detecta e modifica os args baseado no tipo
            -- MM2 envia: FireServer(CFrame_origin, CFrame_dest)  ou  FireServer(Vector3)
            -- Modificamos conforme o modo
            for i, v in ipairs(args) do
                if typeof(v) == "CFrame" then
                    if i == 1 and (mode == "Start" or mode == "Both") then
                        -- primeiro CFrame = origin → coloca no murder
                        args[i] = mCF or CFrame.new(mPos)
                    elseif i == 2 and (mode == "End" or mode == "Both") then
                        -- segundo CFrame = destino → coloca no murder
                        args[i] = CFrame.new(mPos)
                    end
                elseif typeof(v) == "Vector3" then
                    if mode == "End" or mode == "Both" then
                        args[i] = mPos
                    end
                end
            end

            return origFire(self, unpack(args))
        end))
    end)

    if not ok then
        -- fallback: sem hookfunction, usa Activated como antes
        _bulletTPConns[gun] = gun.Activated:Connect(function()
            if not Config.BulletTP then return end
            local mode = Config.BulletTPMode or "End"
            if mode == "Start" or mode == "Both" then
                local mPos, mCF = GetMurderPos()
                if not mPos then return end
                local char = LocalPlayer.Character
                local hrp  = char and char:FindFirstChild("HumanoidRootPart")
                if not hrp then return end
                local origCF = hrp.CFrame
                hrp.CFrame = mCF or CFrame.new(mPos)
                RunService.Stepped:Wait()
                hrp.CFrame = origCF
            end
        end)
    end
end

local _grabCooldown = false

local function FindGunDrop()
    -- Procura GunDrop em qualquer nível do workspace (ex: workspace.House2.GunDrop)
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj.Name == "GunDrop" and obj:IsA("BasePart") then
            return obj
        end
    end
    return nil
end

local function PerformGrabGun(silent)
    if _grabCooldown then return end
    local char = LocalPlayer.Character
    if not char then if not silent then N("GRAB GUN","Sem personagem!","#FF4444") end; return end
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then if not silent then N("GRAB GUN","Sem HumanoidRootPart!","#FF4444") end; return end

    local gunDrop = FindGunDrop()
    if not gunDrop then
        if not silent then N("GRAB GUN","GunDrop não encontrado!","#FFA500") end
        return
    end

    _grabCooldown = true
    -- Fix: NÃO move TweenFloatPart nem o root (câmera não trava)
    -- Dispara firetouchinterest direto no GunDrop — servidor valida por proximidade
    -- mas o touch é suficiente se o GunDrop tiver TouchTransmitter ativo
    task.spawn(function()
        pcall(firetouchinterest, root, gunDrop, 0)
        pcall(firetouchinterest, root, gunDrop, 1)
        -- ProximityPrompt como fallback extra
        local pp = gunDrop:FindFirstChildOfClass("ProximityPrompt")
            or (gunDrop.Parent and gunDrop.Parent:FindFirstChildOfClass("ProximityPrompt"))
        if pp then pcall(fireproximityprompt, pp) end
        if not silent then N("GRAB GUN","Ativou touch no GunDrop!","#00AAFF") end
        task.delay(0.6, function() _grabCooldown = false end)
    end)
end

do

        if _autoGrabConn1 then _autoGrabConn1:Disconnect() end

    _setupAutoGrab()
end

-- Equipa a faca se estiver na mochila. Retorna true se equipou do backpack (para devolver depois).
local function EquipKnife()
    local char = LocalPlayer.Character
    if not char then return false end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then return false end
    for _, t in pairs(char:GetChildren()) do
        if t:IsA("Tool") and (t.Name:lower():find("knife") or t.Name:lower():find("fac")) then
            return false -- já na mão
        end
    end
    for _, t in pairs(LocalPlayer.Backpack:GetChildren()) do
        if t:IsA("Tool") and (t.Name:lower():find("knife") or t.Name:lower():find("fac")) then
            hum:EquipTool(t); task.wait(0.08); return true
        end
    end
    return false
end

local function GetKnifeRemote()
    local char = LocalPlayer.Character
    if not char then return nil end
    for _, tool in pairs(char:GetChildren()) do
        if tool:IsA("Tool") and (tool.Name:lower():find("knife") or tool.Name:lower():find("fac")) then
            local events = tool:FindFirstChild("Events")
            if events then
                local remote = events:FindFirstChild("KillPlayer") or events:FindFirstChild("Kill")
                if remote then return remote end
            end
        end
    end
    return nil
end

local function KillAll()
    local char = LocalPlayer.Character
    if not char then N("KILL ALL","Sem personagem!","#FF4444"); return end
    -- equipa faca automaticamente se estiver na mochila
    local fromBackpack = EquipKnife()
    char = LocalPlayer.Character
    local knife = nil
    for _, tool in pairs(char:GetChildren()) do
        if tool:IsA("Tool") and (tool.Name:lower():find("knife") or tool.Name:lower():find("fac")) then
            knife = tool; break
        end
    end
    if not knife then N("KILL ALL","Faca não encontrada no personagem ou mochila!","#FF4444"); return end

    local killed = 0
    for _, enemy in pairs(Players:GetPlayers()) do
        if enemy ~= LocalPlayer and enemy.Character then
            local eRoot = enemy.Character:FindFirstChild("HumanoidRootPart")
            local eHum  = enemy.Character:FindFirstChild("Humanoid")
            local eRole = RoleCache[enemy.Name]
            if eRoot and eHum and eHum.Health > 0 and eRole ~= "Murderer" and eRole ~= "Lobby" then
                eRoot.CFrame = knife.Handle.CFrame
                eRoot.AssemblyLinearVelocity = Vector3.zero
                eRoot.AssemblyAngularVelocity = Vector3.zero
                local remote = GetKnifeRemote()
                if remote then pcall(function() remote:FireServer(eRoot) end) end
                killed = killed + 1
                if Config.KnifeLegitMode then break end
            end
        end
    end

    if killed > 0 then N("KILL ALL","Tentou matar "..killed.." jogadores.","#00FF88")
    else N("KILL ALL","Nenhum alvo válido encontrado.","#FFA500") end
    -- devolve faca pra mochila se foi equipada aqui
    if fromBackpack and knife and knife.Parent == char then knife.Parent = LocalPlayer.Backpack end
end

-- AUTO SHOOT MURDER (wallcheck + range + silent aim melhorado)
-- ============================================================

local function ApplyAutoShootMurder(state)
    if autoShootConn then autoShootConn:Disconnect(); autoShootConn = nil end
    if not state then return end

    local lastShot = 0
    autoShootConn = RunService.Heartbeat:Connect(function()
        if not Config.AutoShootMurder then return end
        local myRole = RoleCache[LocalPlayer.Name]
        if myRole ~= "Sheriff" and myRole ~= "Hero" then return end

        local myChar = LocalPlayer.Character
        local myHRP  = myChar and myChar:FindFirstChild("HumanoidRootPart")
        if not myHRP then return end

        -- delay entre tiros
        if tick() - lastShot < Config.AutoShootDelay then return end

        -- acha murderer
        local targetPlr = nil
        for plrName, role in pairs(RoleCache) do
            if role == "Murderer" then
                local p = Players:FindFirstChild(plrName)
                if p and p.Character then
                    local ph = p.Character:FindFirstChild("Humanoid")
                    local pr = p.Character:FindFirstChild("HumanoidRootPart")
                    if ph and ph.Health > 0 and pr then
                        -- range check
                        local dist = (myHRP.Position - pr.Position).Magnitude
                        if dist <= Config.AutoShootRange then
                            -- wallcheck
                            if Config.WallCheck then
                                if not HasLineOfSight(Camera.CFrame.Position, p.Character, pr.Position) then
                                    targetPlr = nil; break
                                end
                            end
                            targetPlr = p
                        end
                    end
                end
                break
            end
        end

        if not targetPlr then return end
        lastShot = tick() -- marca antes de spawn para evitar spam

        -- Fix: task.spawn para não bloquear o Heartbeat
        -- (o task.wait dentro do Heartbeat causava o personagem desaparecer da tela)
        task.spawn(function()
            local gun = FindGun()
            if not gun then return end
            local char = LocalPlayer.Character
            if not char then return end

            local wasInBackpack = (gun.Parent == LocalPlayer.Backpack)
            if wasInBackpack then
                char.Humanoid:EquipTool(gun)
                task.wait(0.05)
            end

            local hrp = targetPlr.Character and targetPlr.Character:FindFirstChild("HumanoidRootPart")
            if not hrp then
                if wasInBackpack and gun.Parent == char then gun.Parent = LocalPlayer.Backpack end
                return
            end
            local tPos = GetLeadShotPosition(hrp)
            if gun:FindFirstChild("Shoot") then
                pcall(function() gun.Shoot:FireServer(
                    CFrame.new(Camera.CFrame.Position, tPos),
                    CFrame.new(tPos)
                ) end)
                -- Notificação com debounce de 2s para não spammar
                N("AUTO SHOOT", "Atirou em "..targetPlr.Name, "#FF4444", 1)
            end
            if wasInBackpack and gun.Parent == char then gun.Parent = LocalPlayer.Backpack end
        end)
    end)
end

                    if pHRP and pHum and pHum.Health > 0 then

                        if dist <= Config.KnifeTriggerbotDist then
                            -- dispara stab via remote ou firetouchinterest
                            local stab = knife:FindFirstChild("Stab")

                                pcall(function() stab:FireServer("Slash") end)
                            else
                                local remote = GetKnifeRemote()
                                if remote then pcall(function() remote:FireServer(pHRP) end) end
                            end
                            lastStab = tick()
                            break
                        end
                    end
                end
            end
        end
    end)
end

-- ============================================================
-- FLING (sistema de 3 connections — baseado no mm2/mm2_fix1)
-- Usa Velocity+RotVelocity via Heartbeat/RenderStepped — mais
-- confiável que AssemblyLinearVelocity+BodyVelocity
-- ============================================================
local _flingFL  = false
local _flingOri = nil
local _flingRet = false
local _flingTog = true
local _flingC1, _flingC2, _flingC3 = nil, nil, nil

local function _flingStop()
    _flingFL = false
    if _flingC1 then _flingC1:Disconnect(); _flingC1 = nil end
    if _flingC2 then _flingC2:Disconnect(); _flingC2 = nil end
    if _flingC3 then _flingC3:Disconnect(); _flingC3 = nil end
    local char = LocalPlayer.Character
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then workspace.CurrentCamera.CameraSubject = hum end
    end
end

local function _flingBack()
    if _flingRet then return end
    _flingRet = true
    local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if hrp and _flingOri then
        hrp.Velocity = Vector3.zero
        task.wait()
        hrp.CFrame = CFrame.new(_flingOri)
    end
    _flingStop()
end

local function FlingPlayer(targetPlayer)
    local char = targetPlayer.Character
    if not char then flingInProgress = false; return end
    local tgt  = char:FindFirstChild("HumanoidRootPart")
    local tHum = char:FindFirstChildOfClass("Humanoid")
    if not tgt or not tHum then flingInProgress = false; return end
    if tHum.Sit then flingInProgress = false; return end

    _flingStop()
    _flingFL  = true
    _flingRet = false
    flingInProgress = true

    local myChar = LocalPlayer.Character
    local me = myChar and myChar:FindFirstChild("HumanoidRootPart")
    if not me then flingInProgress = false; return end
    _flingOri = me.Position

    workspace.CurrentCamera.CameraSubject = tHum

    -- FallenPartsDestroyHeight = NaN — impede sumida no void
    local origFPDH = workspace.FallenPartsDestroyHeight
    workspace.FallenPartsDestroyHeight = 0/0

    -- C1: mantém nosso HRP caindo rápido para forçar contato
    _flingC1 = RunService.Heartbeat:Connect(function()
        if not _flingFL then return end
        local h = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if h then
            h.Velocity = Vector3.new(0, -9e7, 0)
            RunService.RenderStepped:Wait()
            h.Velocity = Vector3.new(0, -9e7, 0)
            RunService.Stepped:Wait()
            h.Velocity = Vector3.new(0, -2, 0)
        end
    end)

    -- C2: posiciona nosso HRP ao lado do alvo a cada RenderStepped
    _flingC2 = RunService.RenderStepped:Connect(function()
        if not _flingFL then return end
        local th = char:FindFirstChild("HumanoidRootPart")
        local h  = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if th and h then
            local off = _flingTog and 0.3 or -0.3
            _flingTog = not _flingTog
            local dir = th.CFrame.LookVector * off
            local pos = Vector3.new(
                th.Position.X + th.Velocity.X * 0.63 + dir.X,
                th.Position.Y,
                th.Position.Z + th.Velocity.Z * 0.63 + dir.Z
            )
            h.CFrame = CFrame.new(pos) * CFrame.Angles(0, math.rad(90), 0)
        end
    end)

    -- C3: detecta fling bem-sucedido (vel > 500) e volta
    _flingC3 = RunService.Heartbeat:Connect(function()
        if not _flingFL then return end
        local th  = char:FindFirstChild("HumanoidRootPart")
        local tH2 = char:FindFirstChildOfClass("Humanoid")
        if (not tH2 or tH2.Health <= 0) or (th and th.Velocity.Magnitude > 500) then
            workspace.FallenPartsDestroyHeight = origFPDH
            _flingBack()
            flingInProgress = false
        end
    end)

    -- Timeout de 5s
    task.delay(5, function()
        if not _flingRet then
            workspace.FallenPartsDestroyHeight = origFPDH
            _flingBack()
            flingInProgress = false
        end
    end)
end

local function FlingAllLoop()
    flingInProgress = true
    while flingAllActive do
        -- Recalcula a cada volta (jogador pode sair)
        local players = {}
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Character
              and p.Character:FindFirstChild("Humanoid")
              and p.Character.Humanoid.Health > 0 then
                table.insert(players, p)
            end
        end
        if #players == 0 then task.wait(1); continue end
        for _, t in ipairs(players) do
            if not flingAllActive then break end
            if t and t.Parent == Players and t.Character then
                flingActive = true; flingTargetPlayer = t
                FlingPlayer(t)
                -- espera o fling terminar (max 6s)
                local w = 0
                while flingInProgress and w < 6 do task.wait(0.1); w = w + 0.1 end
                flingActive = false
            end
            task.wait(0.5)
        end
    end
    flingInProgress = false
end

local function StopFling()
    flingActive=false; flingAllActive=false
    if flingThread then task.cancel(flingThread); flingThread=nil end
    Config.FlingSheriff=false; Config.FlingMurderer=false; Config.FlingHero=false; Config.FlingAll=false
    flingInProgress=false
    -- Para as connections do novo sistema de fling
    pcall(_flingStop)
end

local function StartFling(target)
    if flingInProgress then N("FLING","Aguarde o fling atual!","#FFA500"); return end
    if flingThread then task.cancel(flingThread); flingThread=nil end
    flingActive=false; flingAllActive=false
    if target == "all" then
        flingAllActive=true; flingThread=task.spawn(FlingAllLoop)
    else
        local targetPlayer
        if type(target)=="string" then
            for plrName, role in pairs(RoleCache) do
                if role==target then targetPlayer=Players:FindFirstChild(plrName); break end
            end
        else targetPlayer=target end
        if targetPlayer then flingActive=true; flingTargetPlayer=targetPlayer; flingThread=task.spawn(FlingPlayer, targetPlayer) end
    end
end

local function ToggleFling(roleName)
    PlayClickSound()
    if flingInProgress then N("FLING","Aguarde o fling atual!","#FFA500"); return end
    local active = (roleName=="Sheriff" and Config.FlingSheriff) or
                   (roleName=="Murderer" and Config.FlingMurderer) or
                   (roleName=="Hero" and Config.FlingHero)
    StopFling()
    if not active then
        StartFling(roleName)
        if roleName=="Sheriff"  then Config.FlingSheriff=true
        elseif roleName=="Murderer" then Config.FlingMurderer=true
        elseif roleName=="Hero" then Config.FlingHero=true end
    end
end

local function ToggleFlingAll()
    PlayClickSound()
    if flingInProgress then N("FLING","Aguarde o fling atual!","#FFA500"); return end
    if Config.FlingAll then StopFling()
    else StopFling(); StartFling("all"); Config.FlingAll=true end
end

local function ToggleFlingPlayer(playerName)
    PlayClickSound()
    if flingInProgress then N("FLING","Aguarde o fling atual!","#FFA500"); return end
    local plr = Players:FindFirstChild(playerName)
    if not plr then N("FLING","Player não encontrado!","#FF4444"); return end
    if flingActive and flingTargetPlayer == plr then StopFling()
    else StopFling(); StartFling(plr) end
end

        local char = LocalPlayer.Character
        if char then
            local root = char:FindFirstChild("HumanoidRootPart")
            if root then
                local myRole = RoleCache[LocalPlayer.Name]
                if myRole == "Murderer" then
                    local remote = GetKnifeRemote()
                    if remote then
                        for _, plr in ipairs(Players:GetPlayers()) do
                            if plr ~= LocalPlayer and plr.Character then
                                local pRoot = plr.Character:FindFirstChild("HumanoidRootPart")
                                local pHum  = plr.Character:FindFirstChild("Humanoid")
                                if pRoot and pHum and pHum.Health > 0 then
                                    local dist = (root.Position - pRoot.Position).Magnitude
                                    if dist <= Config.KnifeAuraDist then
                                        pcall(function() remote:FireServer(pRoot) end)
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end

    -- Knife Homing
    if Config.KnifeHoming then
        local char = LocalPlayer.Character
        if char then
            local myRole = RoleCache[LocalPlayer.Name]
            if myRole == "Murderer" then
                for _, knifePart in ipairs(Workspace:GetDescendants()) do
                    if knifePart.Name == "Knife_Handle" and knifePart:IsA("BasePart") then
                        local closest, closestDist = nil, math.huge
                        for _, plr in ipairs(Players:GetPlayers()) do
                            if plr ~= LocalPlayer and plr.Character then
                                local pRoot = plr.Character:FindFirstChild("HumanoidRootPart")
                                local pHum  = plr.Character:FindFirstChild("Humanoid")
                                if pRoot and pHum and pHum.Health > 0 then
                                    local dist = (knifePart.Position - pRoot.Position).Magnitude
                                    if dist < closestDist then
                                        closestDist = dist; closest = pRoot
                                    end
                                end
                            end
                        end
                        if closest and closestDist <= Config.KnifeTriggerDist then
                            local dir = (closest.Position - knifePart.Position).Unit
                            knifePart.AssemblyLinearVelocity = dir * 100
                        end
                    end
                end
            end
        end
    end

    -- WalkSpeed / JumpPower
    local char = LocalPlayer.Character
    if char then
        local hum = char:FindFirstChild("Humanoid")
        if hum then
            if Config.WalkSpeedEnabled then hum.WalkSpeed = Config.WalkSpeedValue end
            if Config.JumpPowerEnabled  then hum.JumpPower = Config.JumpPowerValue end
        end
    end

    -- Anti-Fling
    if Config.AntiFling then
        local c = LocalPlayer.Character
        if c then
            local root = c:FindFirstChild("HumanoidRootPart")
            if root and root.AssemblyLinearVelocity.Magnitude > 500 then
                root.AssemblyLinearVelocity = Vector3.zero
                root.AssemblyAngularVelocity = Vector3.zero
            end
        end
    end
end)

task.spawn(function()
    while true do
        task.wait(1)
        local roles = nil

        if not playerDataRemote then
            playerDataRemote = ReplicatedStorage:FindFirstChild("GetPlayerData", true)
        end
        if playerDataRemote then
            local ok, data = pcall(function() return playerDataRemote:InvokeServer() end)
            if ok and data then
                local nr = {}; local hasMurder = false
                for plr, pd in pairs(data) do
                    nr[plr] = (pd.Dead or not pd.Role or pd.Role=="") and "Lobby" or pd.Role
                    if pd.Role == "Murderer" then hasMurder = true end
                end
                if not hasMurder then rolesNotified = false end
                roles = nr
            end
        end

        if roles then
            local myRole = roles[LocalPlayer.Name]
            if myRole and myRole ~= lastMyRole and myRole ~= "Lobby" then
                lastMyRole = myRole
                if Config.NotifyRoles then
                    N("SEU PAPEL", "Você é: "..myRole, "#00AAFF", 5)
                end
                if Config.InstantRole then
                    N("⚡ INSTANT ROLE", "Você é: "..string.upper(myRole), "#FFD700", 5)
                end
            end

            -- Detect murderer/gun for notifications
            for plrName, role in pairs(roles) do
                local old = RoleCache[plrName]
                if old ~= role then
                    if Config.NotifyRoles and (role=="Murderer" or role=="Sheriff") then
                        local plr = Players:FindFirstChild(plrName)
                        N("ROLE ALERT", plrName.." é "..role, role=="Murderer" and "#FF4444" or "#4488FF", 4)
                    end
                end
            end
            RoleCache = roles
        end

        -- Notify Gun Pickup / Drop
        if Config.NotifyGunPickup or Config.NotifyGunDrop then
            for _, plr in ipairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer then
                    local hasGun = false
                    if plr.Character then
                        for _, t in ipairs(plr.Character:GetChildren()) do
                            if t:IsA("Tool") and (t.Name:lower():find("gun") or t.Name=="Revolver") then
                                hasGun = true; break
                            end
                        end
                    end
                    -- simple pickup/drop detection per-player would require per-player state tracking
                    -- kept minimal here for performance
                end
            end
        end
    end
end)

local function AutoFarmCoinsFunc()
    local function tp(target)
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("HumanoidRootPart") then
            char.HumanoidRootPart.CFrame = target.CFrame * CFrame.new(0,3,0)
        end
    end
    local function getNearest()
        local char = LocalPlayer.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
        local root = char.HumanoidRootPart
        local nearest, nearestDist = nil, math.huge
        for _, obj in pairs(Workspace:GetDescendants()) do
            if obj.Name=="Coin_Server" and obj:IsA("BasePart") then
                local dist = (obj.Position - root.Position).Magnitude
                if dist < nearestDist then nearestDist=dist; nearest=obj end
            end
        end
        return nearest
    end

    while autoFarmCoinsRunning and task.wait(0.1) do
        local char = LocalPlayer.Character
        if not char or not char:FindFirstChild("Humanoid") or char.Humanoid.Health<=0 then
            autoFarmCoinsRunning=false; Config.AutoFarmCoins=false; break
        end
        local target = getNearest()
        if target then
            tp(target)
            local v = target:FindFirstChild("CoinVisual")
            while v and not v:GetAttribute("Collected") and v.Parent do
                char = LocalPlayer.Character
                if not char or not char:FindFirstChild("Humanoid") or char.Humanoid.Health<=0 then
                    autoFarmCoinsRunning=false; break
                end
                local n = getNearest()
                if n and n ~= target then break end
                task.wait(0.1)
            end
        else task.wait(0.5) end
    end
end

local function DodgeKnife_RemoveAll()
    for _, k in ipairs(workspace:GetChildren()) do
        if k:IsA("Model") and k.Name == "ThrowingKnife" then k:Destroy() end
    end
end

local function DodgeKnife_IsPathClear(startPos, endPos)
    local rp = RaycastParams.new()
    rp.FilterType = Enum.RaycastFilterType.Exclude
    rp.FilterDescendantsInstances = {LocalPlayer.Character}
    return not workspace:Raycast(startPos, (endPos - startPos), rp)
end

local function DodgeKnife_TeleportAway(knife)
    if not (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")) then return end
    if LocalPlayer.Backpack:FindFirstChild("Knife") or LocalPlayer.Character:FindFirstChild("Knife") then return end
    local playerPos = LocalPlayer.Character.HumanoidRootPart.Position
    local blade = knife:FindFirstChild("BladePosition")
    if blade and blade:IsA("BasePart") then
        local kpos = blade.Position
        local dist = (playerPos - kpos).Magnitude
        if dist <= 9 then
            local dir = (playerPos - kpos).Unit
            local perp = Vector3.new(-dir.Z, 0, dir.X)
            local safe = kpos + perp * 9
            local target = Vector3.new(safe.X, playerPos.Y, safe.Z)
            if DodgeKnife_IsPathClear(playerPos, target) then
                if TweenFloatPart then
                    tweenMoving = false
                    TweenFloatPart.CFrame = CFrame.new(target)
                else
                    LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(target)
                end
            end
        end
    end
end

local function ApplyDodgeKnife(state)
    if dodgeKnifeDescConn then dodgeKnifeDescConn:Disconnect(); dodgeKnifeDescConn = nil end
    if dodgeKnifeHBConn   then dodgeKnifeHBConn:Disconnect();   dodgeKnifeHBConn   = nil end
    activeKnives = {}
    if state then
        DodgeKnife_RemoveAll()
        dodgeKnifeDescConn = workspace.DescendantAdded:Connect(function(k)
            if k:IsA("Model") and k.Name == "ThrowingKnife" then
                    table.insert(activeKnives, k)
                    k.Destroying:Connect(function()
                        for i = #activeKnives, 1, -1 do
                            if activeKnives[i] == k then table.remove(activeKnives, i); break end
                        end
                    end)
                end
        end)
        dodgeKnifeHBConn = RunService.Heartbeat:Connect(function()
            for _, knife in ipairs(activeKnives) do
                if knife and knife.Parent then
                    local blade = knife:FindFirstChild("BladePosition")
                    if blade and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                        local dist = (LocalPlayer.Character.HumanoidRootPart.Position - blade.Position).Magnitude
                        if dist <= 25 then DodgeKnife_TeleportAway(knife) end
                    end
                end
            end
        end)
    end
end

local function ApplyStabReach(state)
    if stabReachHBConn then stabReachHBConn:Disconnect(); stabReachHBConn = nil end
    if not state then return end
    stabReachHBConn = RunService.Heartbeat:Connect(function()
        if not Config.StabReach then return end

        local root  = char:FindFirstChild("HumanoidRootPart")
        local knife = char:FindFirstChild("Knife")
        if not (root and knife) then return end
        local stab = knife:FindFirstChild("Stab")
        if not stab then return end
        local closest, shortDist = nil, Config.StabReachDist
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character then
                local tr = plr.Character:FindFirstChild("HumanoidRootPart")
                local th = plr.Character:FindFirstChild("Humanoid")
                if tr and th and th.Health > 0 then
                    local d = (root.Position - tr.Position).Magnitude
                    if d <= Config.StabReachDist and d < shortDist then
                        closest = plr; shortDist = d
                    end
                end
            end
        end
        if closest then
            local tr = closest.Character:FindFirstChild("HumanoidRootPart")
            if tr then
                pcall(function() stab:FireServer("Slash") end)
                pcall(function() firetouchinterest(tr, knife.Handle, 1) end)
                pcall(function() firetouchinterest(tr, knife.Handle, 0) end)
            end
        end
    end)
end

local function ApplySpeedGlitch(state)
    if speedGlitchConn then speedGlitchConn:Disconnect(); speedGlitchConn = nil end
    speedGlitchAccel = 0.3
    if not state then return end
    speedGlitchConn = RunService.Heartbeat:Connect(function()
        if not Config.SpeedGlitch then return end
        local char = LocalPlayer.Character
        if not char then return end
        local hum  = char:FindFirstChildOfClass("Humanoid")
        local hrp  = char:FindFirstChild("HumanoidRootPart")
        if not (hum and hrp) then return end
        local hasItem = false
        for _, t in ipairs(char:GetChildren()) do
            if t:IsA("Tool") then hasItem = true; break end
        end
        local shouldRun = (Config.SpeedGlitchMode == "With Item" and hasItem) or
                          (Config.SpeedGlitchMode == "No Item"   and not hasItem)
        if not shouldRun then speedGlitchAccel = 0.3; return end
        local hstate = hum:GetState()
        if hstate == Enum.HumanoidStateType.Jumping or hstate == Enum.HumanoidStateType.Freefall then
            local dir = hum.MoveDirection
            if dir.Magnitude > 0 then
                speedGlitchAccel = math.min(speedGlitchAccel + 0.05, 1)
                local boost = Config.SpeedGlitchPower * speedGlitchAccel
                char:TranslateBy(dir.Unit * boost / 150)
            end
        else
            speedGlitchAccel = 0.3
        end
    end)
end

local function TweenFarm_IsBagFull()
    local ok, result = pcall(function()
        local gui = LocalPlayer:FindFirstChild("PlayerGui")
        if not gui then return false end
        local mg = gui:FindFirstChild("MainGUI")
        if not mg then return false end
        -- percorre todos os descendentes do MainGUI procurando Dock > CoinBags > Container > Coin > Full
        for _, dk in ipairs(mg:GetDescendants()) do
            if dk.Name == "Dock" then
                local bg = dk:FindFirstChild("CoinBags")
                if bg then
                    local ct = bg:FindFirstChild("Container")
                    if ct then
                        local co = ct:FindFirstChild("Coin")
                        if co then
                            local fl = co:FindFirstChild("Full")
                            if fl and fl:IsA("GuiObject") then
                                return fl.Visible
                            end
                        end
                    end
                end
            end
        end
        return false
    end)
    return ok and result or false
end

local function TweenFarm_GetCoins()
    local coins = {}
    if tweenCurrentContainer then
        for _, c in ipairs(tweenCurrentContainer:GetChildren()) do
            if c:IsA("BasePart") and not tweenVisited[c] then
                c.Size = Vector3.new(5,6,5); table.insert(coins, c)
            end
        end
    end
    return coins
end

local function TweenFarm_GetNext()
    local coins = TweenFarm_GetCoins()
    if Config.TweenRandomCoin then
        local cands = {}
        for _, c in ipairs(coins) do
            if TweenFloatPart and (c.Position - TweenFloatPart.Position).Magnitude <= 80 then
                table.insert(cands, c)
            end
        end
        return #cands > 0 and cands[math.random(#cands)] or nil
    else
        local best, bd
        for _, c in ipairs(coins) do
            if TweenFloatPart then
                local d = (c.Position - TweenFloatPart.Position).Magnitude
                if not best or d < bd then best, bd = c, d end
            end
        end
        return best, bd
    end
end

local function TweenFarm_MoveNext()
    if not TweenAutoFarmEnabled or not TweenFloatPart then return end
    do
        local char = LocalPlayer.Character
        local hum  = char and char:FindFirstChildOfClass("Humanoid")
        if not (char and hum and hum.Health > 0) then tweenMoving = false; return end
    end

    if TweenFarm_IsBagFull() then
        if Config.TweenResetBag then
            local h = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
            if h then h.Health = 0 end
            task.wait(3) -- espera respawn
        end
        -- bag cheia: pausa o tween mas mantém TweenFloatPart vivo
        tweenMoving = false
        return
    end

    tweenMoving = true
    if not TweenFloatPart then tweenMoving = false; return end
    local coin, dist = TweenFarm_GetNext()
    if not coin then tweenVisited = {}; tweenMoving = false; return end

    tweenVisited[coin] = true
    local offsetY = (Config.TweenFarmMode == "Below the map") and -3.3 or 1
    local target  = coin.Position + Vector3.new(0, offsetY, 0)
    dist = dist or (TweenFloatPart and (target - TweenFloatPart.Position).Magnitude or 0)

    if dist > 90 then
        if tick() - tweenLastTeleport >= 3 then
            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if hrp then hrp.CFrame = CFrame.new(target); tweenLastTeleport = tick() end
        end
        task.wait(Config.TweenCoinDelay / 10)
        tweenMoving = false
        return
    end

    if TweenFloatPart then
        local tw = TweenService:Create(TweenFloatPart,
            TweenInfo.new(dist / Config.TweenSpeed, Enum.EasingStyle.Linear),
            { Position = target })
        tw.Completed:Connect(function()
            task.wait(Config.TweenCoinDelay / 10)
            tweenMoving = false
        end)
        tw:Play()
    else tweenMoving = false end
end

-- TWEEN FARM — KILL MURDER (bag full → TP 5 studs perto do Murder → atira com aim → volta)
-- ============================================================
local tweenKillSheriffThread = nil
local tweenKillSheriffRunning = false

local function TweenKillSheriff_Stop()
    tweenKillSheriffRunning = false
    if tweenKillSheriffThread then task.cancel(tweenKillSheriffThread); tweenKillSheriffThread = nil end
end

local function TweenKillSheriff_Loop()
    tweenKillSheriffRunning = true
    while tweenKillSheriffRunning and Config.TweenKillSheriff do
        task.wait(1)
        if not Config.TweenKillSheriff then break end
        if not TweenFarm_IsBagFull() then continue end
        task.wait(0.5) -- pequena pausa antes de agir

        -- acha o Murderer
        local murderPlr = nil
        for plrName, role in pairs(RoleCache) do
            if role == "Murderer" then murderPlr = Players:FindFirstChild(plrName); break end
        end
        if not (murderPlr and murderPlr.Character and murderPlr.Character:FindFirstChild("HumanoidRootPart")) then continue end

        local myChar = LocalPlayer.Character
        local myHRP  = myChar and myChar:FindFirstChild("HumanoidRootPart")
        if not myHRP then continue end
        -- FloatPart pode não existir se TweenAutoFarm estiver desligado
        if not TweenFloatPart then
            -- fallback: TP direto no HRP sem tween ativo
            local origCF2   = myHRP.CFrame
            local murderHRP2 = murderPlr.Character.HumanoidRootPart
            local dir2   = (murderHRP2.CFrame.LookVector * -1).Unit
            myHRP.CFrame = CFrame.new(murderHRP2.Position + dir2 * 5, murderHRP2.Position)
            task.wait(0.1)
            local gun2 = nil; local wib2 = false
            for _, i in ipairs(myChar:GetChildren()) do
                if i:IsA("Tool") and (string.find(string.lower(i.Name),"gun") or i.Name=="Revolver") then gun2=i; break end
            end
            if not gun2 then
                for _, i in ipairs(LocalPlayer.Backpack:GetChildren()) do
                    if i:IsA("Tool") and (string.find(string.lower(i.Name),"gun") or i.Name=="Revolver") then gun2=i; wib2=true; break end
                end
                if gun2 then myChar.Humanoid:EquipTool(gun2); task.wait(0.1) end
            end
            if gun2 then
                local tPos2 = GetLeadShotPosition(murderHRP2)
                if gun2:FindFirstChild("Shoot") then
                    pcall(function() gun2.Shoot:FireServer(CFrame.new(Camera.CFrame.Position, tPos2), CFrame.new(tPos2)) end)
                end
                if wib2 and gun2.Parent == myChar then gun2.Parent = LocalPlayer.Backpack end
            end
            task.wait(0.15)
            if myHRP and myHRP.Parent then myHRP.CFrame = origCF2 end
            N("KILL MURDER", "Tiro no Murder! Voltando...", "#FF4444", 2)
            continue
        end

        -- TweenAutoFarm ativo: o Heartbeat força hrp.CFrame = TweenFloatPart.CFrame a cada frame.
        -- Solução: mover o TweenFloatPart (não o HRP) → Heartbeat sincroniza automaticamente.
        local origFloatCF = TweenFloatPart.CFrame  -- salva posição atual do farm
        tweenMoving = false                          -- cancela tween em curso

        local murderHRP = murderPlr.Character.HumanoidRootPart

        -- Move FloatPart 5 studs na frente do Murder → Heartbeat vai mover HRP junto
        local dir5   = (murderHRP.CFrame.LookVector * -1).Unit
        local nearCF = CFrame.new(murderHRP.Position + dir5 * 5, murderHRP.Position)
        TweenFloatPart.CFrame = nearCF
        task.wait(0.15) -- aguarda Heartbeat sincronizar HRP

        -- equipa arma (char primeiro, depois backpack)
        local gun = nil
        local wasInBackpack = false
        for _, i in ipairs(myChar:GetChildren()) do
            if i:IsA("Tool") and (string.find(string.lower(i.Name),"gun") or i.Name=="Revolver") then gun=i; break end
        end
        if not gun then
            for _, i in ipairs(LocalPlayer.Backpack:GetChildren()) do
                if i:IsA("Tool") and (string.find(string.lower(i.Name),"gun") or i.Name=="Revolver") then
                    gun=i; wasInBackpack=true; break
                end
            end
            if gun then myChar.Humanoid:EquipTool(gun); task.wait(0.1) end
        end

        if gun then
            local tPos = GetLeadShotPosition(murderHRP)
            if gun:FindFirstChild("Shoot") then
                pcall(function() gun.Shoot:FireServer(
                    CFrame.new(Camera.CFrame.Position, tPos),
                    CFrame.new(tPos)
                ) end)
            end
            if wasInBackpack and gun.Parent == myChar then gun.Parent = LocalPlayer.Backpack end
        end

        task.wait(0.15)
        -- Volta FloatPart para onde estava → Heartbeat leva HRP de volta e farm continua
        if TweenFloatPart then TweenFloatPart.CFrame = origFloatCF end
        N("KILL MURDER", "Tiro no Murder! Voltando...", "#FF4444", 2)
    end
    tweenKillSheriffRunning = false
end

-- ============================================================
-- TWEEN FARM — KILL ALL (bag full + eu sou murderer → mata todos com faca)
-- ============================================================
local tweenKillAllThread = nil
local tweenKillAllRunning = false

local function TweenKillAll_Stop()
    tweenKillAllRunning = false
    if tweenKillAllThread then task.cancel(tweenKillAllThread); tweenKillAllThread = nil end
end

local function TweenKillAll_Loop()
    tweenKillAllRunning = true
    while tweenKillAllRunning and Config.TweenKillAll do
        task.wait(1)
        if not Config.TweenKillAll then break end
        local myRole = RoleCache[LocalPlayer.Name]
        if myRole ~= "Murderer" then continue end
        if not TweenFarm_IsBagFull() then continue end
        task.wait(0.5) -- pequena pausa antes de agir

        local myChar = LocalPlayer.Character
        local myHRP  = myChar and myChar:FindFirstChild("HumanoidRootPart")
        if not myHRP then continue end

        -- equipa faca se necessário
        local fromBP_ka = EquipKnife()
        local remote = GetKnifeRemote()
        if not remote then
            -- sem faca, não pode matar
            continue
        end

        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character then
                local pRoot = plr.Character:FindFirstChild("HumanoidRootPart")
                local pHum  = plr.Character:FindFirstChild("Humanoid")
                local pRole = RoleCache[plr.Name]
                if pRoot and pHum and pHum.Health > 0 and pRole ~= "Murderer" and pRole ~= "Lobby" then
                    local targetCF = pRoot.CFrame * CFrame.new(0, 0, 2)
                    if TweenFloatPart then
                        -- TweenAutoFarm ativo: move FloatPart → Heartbeat sincroniza HRP
                        local origFloatCF_ka = TweenFloatPart.CFrame
                        tweenMoving = false
                        TweenFloatPart.CFrame = targetCF
                        task.wait(0.08)
                        pcall(function() remote:FireServer(pRoot) end)
                        task.wait(0.12)
                        if TweenFloatPart then TweenFloatPart.CFrame = origFloatCF_ka end
                    else
                        local origCF = myHRP.CFrame
                        myHRP.CFrame = targetCF
                        task.wait(0.05)
                        pcall(function() remote:FireServer(pRoot) end)
                        task.wait(0.1)
                        myHRP.CFrame = origCF
                    end
                    task.wait(0.3)
                end
            end
        end
        -- devolve faca se foi equipada aqui
        if fromBP_ka then
            local char2 = LocalPlayer.Character
            if char2 then
                for _, t in pairs(char2:GetChildren()) do
                    if t:IsA("Tool") and (t.Name:lower():find("knife") or t.Name:lower():find("fac")) then
                        t.Parent = LocalPlayer.Backpack; break
                    end
                end
            end
        end
        N("KILL ALL", "Todos eliminados!", "#FF4444", 2)
    end
    tweenKillAllRunning = false
end

-- TELEPORT HELPERS (role / player)
-- ============================================================
local function GetPlayerByRole(role)
    for plrName, r in pairs(RoleCache) do
        if r == role then return Players:FindFirstChild(plrName) end
    end
    return nil
end

local function TeleportToRole(role)
    local plr = GetPlayerByRole(role)
    if plr and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("HumanoidRootPart") then
            char.HumanoidRootPart.CFrame = plr.Character.HumanoidRootPart.CFrame
            N("TP ROLE", "Teleportado para "..role.." ("..plr.Name..")", "#00AAFF")
        end
    else
        N("TP ROLE", role.." não encontrado!", "#FF4444")
    end
end

local function TeleportToPlayer(mode)
    local target = nil
    if mode == "Closest Player" then
        local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            local closest, cdist = nil, math.huge
            for _, plr in ipairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer and plr.Character then
                    local tr = plr.Character:FindFirstChild("HumanoidRootPart")
                    if tr then
                        local d = (hrp.Position - tr.Position).Magnitude
                        if d < cdist then closest, cdist = plr, d end
                    end
                end
            end
            target = closest
        end
    else
        local list = {}
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer then table.insert(list, plr) end
        end
        if #list > 0 then target = list[math.random(#list)] end
    end
    if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("HumanoidRootPart") then
            char.HumanoidRootPart.CFrame = target.Character.HumanoidRootPart.CFrame
            N("TP PLAYER", "Teleportado para "..target.Name, "#00AAFF")
        end
    else N("TP PLAYER", "Nenhum jogador encontrado!", "#FF4444") end
end

                    coinOrig[obj] = obj.Size; obj.Size = coinOrig[obj] * 4

                end
            end)

            for obj, orig in pairs(coinOrig) do if obj and obj.Parent then obj.Size = orig end end
            if coinReachConn then coinReachConn:Disconnect(); coinReachConn = nil end
            coinOrig = {}
        end
        ScheduleSave()
    end,

        PlayClickSound()
        local ok, data = pcall(function()
            local remote = ReplicatedStorage:FindFirstChild("GetPlayerData", true)
            if remote then return remote:InvokeServer() end
        end)
        if not ok or not data then N("SNITCH","Falhou em obter dados!","#FF4444"); return end
        local sheriff, murderer
        for name, role in pairs(data) do
            if type(role) == "table" then
                if role.Role == "Murderer" then murderer = name
                elseif role.Role == "Sheriff" then sheriff = name end
            end
        end
        local TCS = game:GetService("TextChatService")
        local ch  = TCS and TCS.TextChannels and TCS.TextChannels:FindFirstChild("RBXGeneral")
        if ch then
            if sheriff and murderer then
                pcall(function() ch:SendAsync(sheriff.." is the Sheriff and "..murderer.." is the Murderer.") end)
            elseif murderer then
                pcall(function() ch:SendAsync(murderer.." is the Murderer.") end)
            elseif sheriff then
                pcall(function() ch:SendAsync(sheriff.." is the Sheriff.") end)
            else N("SNITCH","Nenhum papel detectado!","#FFA500"); return end
            N("SNITCH","Enviado no chat!","#00FF88")
        else N("SNITCH","TextChatService não encontrado!","#FF4444") end
    end,
})

local function ApplyAllConfig()
    if Config.AntiLag         then Lighting.GlobalShadows=false; Lighting.ShadowSoftness=0 end
    if Config.CoinsRemove     then ApplyCoinsRemove(true) end
    if Config.OptimizerCoins  then ApplyOptimizerCoins(true) end
    if Config.DisableFootstep then ApplyDisableFootstep(true) end
    if Config.ShowRoundTimer  then ApplyShowRoundTimer(true) end
    if Config.SecondLife      then ApplySecondLife(true) end
    if Config.AutoPrankBomb   then ApplyAutoPrankBomb(true) end
    if Config.MuteRadios      then ApplyMuteRadios(true) end
    if Config.AutoShootMurder then ApplyAutoShootMurder(true) end
    if Config.AutoKillSheriff then ApplyAutoKillSheriff(true) end
    if Config.KnifeTriggerbot then ApplyKnifeTriggerbot(true) end
    if Config.DodgeKnife      then ApplyDodgeKnife(true) end
    if Config.StabReach       then ApplyStabReach(true) end
    if Config.ThrowKnifeReach then ApplyThrowKnifeReach(true) end
    if Config.SpeedGlitch     then ApplySpeedGlitch(true) end

    -- Botões flutuantes
    ShootGui.Enabled    = Config.SilentAim    or false
    GrabFrame.Visible   = Config.ShowGrabBtn  or false
    InvisFrame.Visible  = Config.ShowInvisBtn or false
    DesyncFrame.Visible = Config.ShowDesyncBtn or false

    -- Estado Invisible
    if Config.InvisActive then
        InvisActive = true
        InvisBtn.Text       = "ACTIVE"
        InvisBtn.TextColor3 = Color3.fromRGB(0,255,150)
        InvisStroke.Color   = Color3.fromRGB(0,255,150)
        if LocalPlayer.Character then
            for _, v in pairs(LocalPlayer.Character:GetDescendants()) do
                if (v:IsA("BasePart") or v:IsA("Decal")) and v.Name ~= "HumanoidRootPart" then
                    v.LocalTransparencyModifier = 1
                end
            end
        end
    else
        InvisActive = false
        InvisBtn.Text       = "INVISIBLE"
        InvisBtn.TextColor3 = Color3.fromRGB(140,50,255)
        InvisStroke.Color   = Color3.fromRGB(140,50,255)
    end

    -- Auto Farm Coins
    if Config.AutoFarmCoins then
        autoFarmCoinsRunning = true
        autoFarmCoinsThread  = task.spawn(AutoFarmCoinsFunc)
    end

    -- Tween Auto Farm
    if Config.TweenAutoFarm then
        TweenAutoFarmEnabled = true
        tweenVisited = {}; tweenCurrentContainer = nil
        TweenFarm_CreateFloat()
    end
    if Config.TweenKillSheriff then
        tweenKillSheriffThread = task.spawn(TweenKillSheriff_Loop)
    end
    if Config.TweenKillAll then
        tweenKillAllThread = task.spawn(TweenKillAll_Loop)
    end
    if Config.TweenFlingMurder then
        tweenFlingMurderThread = task.spawn(TweenFlingMurder_Loop)
    end
end

ApplyAllConfig()
